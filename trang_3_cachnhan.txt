    <script>
        function wrapLabels(str, maxWidth) {
            if (str.length <= maxWidth) return [str];
            const words = str.split(' '); let lines = []; let currentLine = words[0];
            for (let i = 1; i < words.length; i++) {
                if ((currentLine + " " + words[i]).length < maxWidth + 1) { currentLine += " " + words[i]; } 
                else { lines.push(currentLine); currentLine = words[i]; }
            }
            lines.push(currentLine); return lines;
        }

        function tooltipTitleCallback(context) {
            const label = context[0].label;
            return Array.isArray(label) ? label.join(' ') : label;
        }

        try {
            const ctx = document.getElementById('placeholderChart').getContext('2d');
            const placeholderLabels = ['Lộc cho Gia Đình', 'Lộc cho Bạn Bè'].map(l => wrapLabels(l, 16));
            new Chart(ctx, {
                type: 'doughnut',
                data: { labels: placeholderLabels, datasets: [{ data: [70, 30], backgroundColor: ['#e52228', '#f59e0b'] }] },
                options: { responsive: true, maintainAspectRatio: false, plugins: { tooltip: { callbacks: { title: tooltipTitleCallback } } } }
            });
        } catch(e) { console.warn("Placeholder chart could not be initialized."); }
    </script>
